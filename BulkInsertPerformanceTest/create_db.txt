-- Type maping tests
CREATE TABLE TEST_TABLE (
    ID INTEGER NOT NULL,
    FIELD1 VARCHAR(100),
    FIELD2 VARCHAR(100),
    FIELD3 VARCHAR(100));

ALTER TABLE TEST_TABLE
ADD CONSTRAINT TEST_TABLE
PRIMARY KEY (ID);


CREATE GENERATOR GEN_TEST_TABLE_ID;

SET TERM ^ ;


/* Trigger: CUSTOMERS_BI */
CREATE OR ALTER TRIGGER TEST_TABLE_BI FOR TEST_TABLE
ACTIVE BEFORE INSERT POSITION 0
as
begin
  if (new.id is null) then
    new.id = gen_id(GEN_TEST_TABLE_ID,1);
end
^

SET TERM ; ^
